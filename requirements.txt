# =============================================================================
# requirements.txt — explicit full dependency list for server deployments
#
# For most developers the recommended install path is:
#   pip install pforge          # lightweight deps auto-installed
#   pforge init --install-gpu-deps             # installs torch + vLLM + training
#
# Use this file when you need a fully explicit install (e.g. Docker, CI, or a
# deployment script that controls the exact install order):
#   pip install -r requirements.txt   (after torch + vLLM are already present)
#
# IMPORTANT — PyTorch and vLLM are NOT listed here and must be installed first
# with the correct CUDA wheel index (pforge init handles this automatically):
#   pip install torch --index-url https://download.pytorch.org/whl/cu121
#   pip install vllm
#
# Do NOT add torch or vllm to this file — vLLM pins its own torch version and
# installing from a plain index URL would overwrite the CUDA-specific wheel.
#
# Version pins are soft minimums (>=) rather than exact pins because:
#   - vLLM pulls in its own transformers/tokenizers pins; letting pip resolve
#     avoids conflicts.
#   - bitsandbytes is CUDA-version-sensitive; exact pins only make sense once
#     you know your CUDA version.
# =============================================================================
